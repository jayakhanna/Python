1    import unittest

3    from Stack import Stack

6    # Unit tests Class to test the Stack class
7    # all the tests are idempotent.
8    class TestStack(unittest.TestCase):
9        testStack = Stack()
10   
11       def test_push(self):
12           try:
13               self.testStack.push('a')
14               self.testStack.push('b')
15               print('Length of stack is:' + str(self.testStack.getlength()))
16           except Exception as e:
17               print(e)
18   
19       def test_pop(self):
20           try:
21               self.testStack.push('b')
22               self.assertEqual('b', self.testStack.pop())
23           except Exception as e:
24               print(e)
25   
26       def test_peek(self):
27           try:
28               self.testStack.push('b')
29               self.testStack.push('a')
30               self.assertEqual(self.testStack.peek(), 'a')
31               self.testStack.pop()
32               self.assertEqual(self.testStack.peek(), 'b')
33           except Exception as e:
34               print(e)
35   
36       def test_emptypop(self):
37           try:
38               self.testStack.stack.clear()
39               fetchItem = self.testStack.pop()
40               self.assertEqual(fetchItem, LookupError.CalcLookupError['EmptyStack'],
41                            "Expected:" + LookupError.CalcLookupError['EmptyStack'] + " Got:" + fetchItem)
42           except Exception as e:
43               print(e)
